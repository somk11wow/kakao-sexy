    $('.dlnk_info').click(function(){

        $('.info_svc').removeClass('info_open');

        if($(this).attr('aria-expanded') == 'false'){
            $(this).parent().addClass('info_open');
            $(this).attr('aria-expanded','true');
            $(this).next().show();
        } else {
            $(this).parent().removeClass('info_open');
            $(this).attr('aria-expanded','false');
            $(this).next().hide();
        }

    });
</script>

<script>
    $(document).ready(function() {
    // 푸터 관련사이트 셀렉트 박스 이벤트

    document.getElementsByClassName('tit_relation')[0].addEventListener('click', function (event) {
        if (this.parentNode.classList.contains('relation_open')) {
                    this.parentNode.classList.remove('relation_open');
                    this.childNodes[0].setAttribute('aria-expanded', 'false');
                } else {
                    this.parentNode.classList.add('relation_open');
                    this.childNodes[0].setAttribute('aria-expanded', 'true');
                }
            });
    });
</script>
